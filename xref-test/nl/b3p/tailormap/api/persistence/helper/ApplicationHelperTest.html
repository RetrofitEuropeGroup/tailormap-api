27  @SpringBootTest(classes = ApplicationHelper.class)
28  class ApplicationHelperTest {
29  
30    @MockBean GeoServiceRepository geoServiceRepository;
31    @MockBean GeoServiceHelper geoServiceHelper;
32    @MockBean FeatureSourceRepository featureSourceRepository;
33    @MockBean ConfigurationRepository configurationRepository;
34    @MockBean ApplicationRepository applicationRepository;
35    @MockBean EntityManager entityManager;
36    @MockBean AuthorizationService authorizationService;
37    @Autowired ApplicationHelper applicationHelper;
38  
39    @Test
40    void testAppWithCrsRD() {
41      Application a = new Application().setCrs("EPSG:28992");
42      MapResponse mr = new MapResponse();
43      applicationHelper.setCrsAndBounds(a, mr);
44      TMCoordinateReferenceSystem crs = mr.getCrs();
45      assertNotNull(crs);
46      assertEquals("EPSG:28992", crs.getCode());
47      assertEquals(StaticTestData.get("RDCrsWkt"), crs.getDefinition());
48      assertEquals("m", crs.getUnit());
49  
50      // initial and max extent should be the CRS envelope
51      assertEquals(a.getInitialExtent(), a.getMaxExtent());
52  
53      Bounds b = mr.getInitialExtent();
54      assertEquals(284300.0254094796, b.getMaxx(), 0.1d, "approximate value should match");
55      assertEquals(636981.7698870874, b.getMaxy(), 0.1d, "approximate value should match");
56      assertEquals(634.5732789819012, b.getMinx(), 0.1d, "approximate value should match");
57      assertEquals(306594.5543000576, b.getMiny(), 0.1d, "approximate value should match");
58    }
59
