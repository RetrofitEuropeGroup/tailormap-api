22  @AutoConfigureMockMvc
23  @PostgresIntegrationTest
24  class PasswordValidationControllerIntegrationTest {
25    @Autowired private MockMvc mockMvc;
26  
27    @Value("${tailormap-api.admin.base-path}")
28    private String adminBasePath;
29  
30    @ParameterizedTest
31    @CsvSource({
32      "short,false",
33      "jI@#%09,false",
34      "jI@#%09y,true",
35      "very long password without special characters but unguessable,true"
36    })
37    @SuppressWarnings("PMD.JUnitTestsShouldIncludeAssert")
38    void test(String password, String expected) throws Exception {
39      mockMvc
40          .perform(
41              post(adminBasePath + "/validate-password")
42                  .contentType(MediaType.APPLICATION_FORM_URLENCODED_VALUE)
43                  .param("password", password)
44                  .accept(MediaType.APPLICATION_JSON))
45          .andExpect(status().isOk())
46          .andExpect(content().contentType(MediaType.APPLICATION_JSON))
47          .andExpect(jsonPath("$.result").value(expected));
48    }
49  }
