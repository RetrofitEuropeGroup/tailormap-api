23  
24  public class WFSFeatureSourceHelper extends FeatureSourceHelper {
25    @Override
26    public DataStore createDataStore(TMFeatureSource tmfs, Integer timeout) throws IOException {
27      Map<String, Object> params = new HashMap<>();
28      if (timeout != null) {
29        params.put(WFSDataStoreFactory.TIMEOUT.key, timeout);
30      }
31  
32      // Params which can not be overridden below
33      String wfsUrl = tmfs.getUrl();
34      if (!wfsUrl.endsWith("&") && !wfsUrl.endsWith("?")) {
35        wfsUrl += wfsUrl.contains("?") ? "&" : "?";
36      }
37      wfsUrl = wfsUrl + "REQUEST=GetCapabilities&SERVICE=WFS";
38      if (!wfsUrl.toUpperCase(Locale.ROOT).contains("VERSION")) {
39        wfsUrl += "&VERSION=1.1.0";
40      }
41  
42      params.put(WFSDataStoreFactory.URL.key, wfsUrl);
43  
44      ServiceAuthentication authentication = tmfs.getAuthentication();
45      if (authentication != null) {
46        if (authentication.getMethod() != ServiceAuthentication.MethodEnum.PASSWORD) {
47          throw new IllegalArgumentException(authentication.getMethod().getValue());
48        }
49        params.put(WFSDataStoreFactory.USERNAME.key, authentication.getUsername());
50        params.put(WFSDataStoreFactory.PASSWORD.key, authentication.getPassword());
51      }
52      return openDatastore(params, WFSDataStoreFactory.PASSWORD.key);
53    }
54  
55    @Override
56    protected TMFeatureTypeInfo getFeatureTypeInfo(
57        TMFeatureType pft, ResourceInfo info, SimpleFeatureSource gtFs) {
58      TMFeatureTypeInfo tmInfo = super.getFeatureTypeInfo(pft, info, gtFs);
59      if (info instanceof FeatureTypeInfo) {
60        FeatureTypeInfo ftInfo = (FeatureTypeInfo) info;
61        tmInfo
62            .schema(info.getSchema())
63            .wgs84BoundingBox(GeoToolsHelper.fromEnvelope(ftInfo.getWGS84BoundingBox()))
64            .defaultSrs(ftInfo.getDefaultSRS())
65            .otherSrs(Set.copyOf(ftInfo.getOtherSRS()))
66            .outputFormats(ftInfo.getOutputFormats())
67            .abstractText(ftInfo.getAbstract());
68      }
69      return tmInfo;
70    }
71  }
