127   @EventListener(ApplicationReadyEvent.class)
128   @Transactional
129   public void populate() throws Exception {
130     InternalAdminAuthentication.setInSecurityContext();
131     try {
132       // Used in conjunction with tailormap-api.database.clean=true so the database has been cleaned
133       // and the latest schema re-created
134       createTestUsersAndGroups();
135       createTestConfiguration();
136     } finally {
137       InternalAdminAuthentication.clearSecurityContextAuthentication();
138     }
139     if (exit) {
140       // Exit after transaction is completed - for 'mvn verify' to populate testdata before
141       // integration tests
142       new Thread(
143               () -> {
144                 try {
145                   Thread.sleep(5000);
146 
147                 } catch (InterruptedException ignored) {
148                   // Ignore
149                 }
150                 SpringApplication.exit(appContext, () -> 0);
151                 System.exit(0);
152               })
153           .start();
154     }
155   }

158     Group groupFoo = new Group().setName("test-foo").setDescription("Used for integration tests.");

162     groupRepository.save(groupBar);
163 
164     Group groupBaz = new Group().setName("test-baz").setDescription("Used for integration tests.");
165     groupRepository.save(groupBaz);
166 
167     // Normal user
168     User u = new User().setUsername("user").setPassword("{noop}user").setEmail("user@example.com");
169     u.getGroups().addAll(List.of(groupFoo, groupBar, groupBaz));
170     userRepository.save(u);
171 
172     // Superuser with all access
173     u = new User().setUsername("tm-admin").setPassword(adminHashedPassword);
174     u.getGroups().add(groupRepository.findById(Group.ADMIN).orElseThrow());
175     userRepository.save(u);
176   }
