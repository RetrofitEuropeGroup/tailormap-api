6   package nl.b3p.tailormap.api.configuration;
7   
8   import java.util.List;
9   import javax.servlet.DispatcherType;
10  import org.springframework.boot.web.servlet.FilterRegistrationBean;
11  import org.springframework.context.annotation.Bean;
12  import org.springframework.context.annotation.Configuration;
13  import org.springframework.core.Ordered;
14  import org.springframework.web.filter.ForwardedHeaderFilter;
15  
16  /** (Servlet) filtering config. */
17  @Configuration
18  public class FilteringConfig {
19  
20    /**
21     * Handle X-Forwarded-for and other proxy headers.
22     *
23     * @return the configured filtering bean
24     */
25    @Bean
26    public FilterRegistrationBean<ForwardedHeaderFilter> forwardedHeaderFilter() {
27      ForwardedHeaderFilter filter = new ForwardedHeaderFilter();
28      FilterRegistrationBean<ForwardedHeaderFilter> registration =
29          new FilterRegistrationBean<>(filter);
30      registration.setDispatcherTypes(
31          DispatcherType.REQUEST, DispatcherType.ASYNC, DispatcherType.ERROR);
32      registration.setOrder(Ordered.HIGHEST_PRECEDENCE);
33      registration.setUrlPatterns(List.of("/*"));
34      return registration;
35    }
36  }
